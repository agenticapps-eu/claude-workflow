export interface SentryOptions {
  dsn: string | undefined;
  environment: string;
  release: string;
  tracesSampleRate: number;
  sendDefaultPii: false;
}

/**
 * Build the options object for `withSentry(optionsFactory, handler)`.
 * ENV-PURE: reads ONLY from `env`, NOT from module-scope singletons.
 *
 * Why env-pure: withSentry's options factory is evaluated per-request BEFORE
 * the inner handler runs. The inner handler is what calls `init(env, ctx)`.
 * Therefore singletons (`serviceName`, `deployEnv`) may still be at default
 * values when this factory runs. Reading directly from `env` matches the
 * pattern openrouter-monitor/src/index.ts:47-57 already uses manually.
 * TRACE_SAMPLE_RATE is a module-scope CONSTANT (baked at scaffold time) so
 * reading it from module scope is fine — it doesn't depend on init() running.
 *
 * Call at the entry-file site:
 *   export default withSentry(env => buildSentryOptions(env), withObservability(handler));
 *
 * See docs/decisions/0034-observability-init-singleton-invariant.md for the
 * runtime model (Cloudflare warm-isolate reuse) that motivates env-purity.
 */
export function buildSentryOptions(env: InitEnv): SentryOptions {
  return {
    dsn: env.{{ENV_VAR_DSN}},
    environment: env.{{ENV_VAR_ENV}} ?? "dev",
    release: env.{{ENV_VAR_SERVICE}} ?? SERVICE_DEFAULT,
    tracesSampleRate: TRACE_SAMPLE_RATE,
    sendDefaultPii: false,
  };
}
