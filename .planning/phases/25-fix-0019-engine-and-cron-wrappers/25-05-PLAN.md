      local stack="$1"
      local baseline="$(stack_template_dir "$stack")/cron-monitor.ts"
      file_canonical_hash "$baseline"

    # ─── All-clean-gate (mirror of 0019's is_known_clean_wrapper, extended for re-rev) ─

    is_clean_to_apply_021() {
      # Returns 0 (clean-to-apply, hash matches v1.19.0), 1 (already-applied,
      # hash matches v1.20.0 AND queue-monitor.ts present), 2 (DIRTY).
      local dir="$1" stack="$2"
      local cm="$dir/$(resolve_anchor_cron_monitor_filename "$dir" "$stack")"
      # NOTE: cron-monitor.ts filename is fixed across stacks; the helper just
      # confirms the file exists at $dir/cron-monitor.ts.
      [ -f "$dir/cron-monitor.ts" ] || return 2
      local hash project; project=$(file_canonical_hash "$dir/cron-monitor.ts") || return 2
      local hash_v_19 hash_v_20
      hash_v_19=$(compute_baseline_hash_v1_19_0 "$stack") || return 2
      hash_v_20=$(compute_baseline_hash_v1_20_0 "$stack") || return 2

      if [ "$project" = "$hash_v_20" ]; then
        # Match v1.20.0 — check queue-monitor.ts presence (twofold idempotency, codex M-8)
        case "$stack" in
          ts-cloudflare-worker|ts-cloudflare-pages)
            if [ -f "$dir/queue-monitor.ts" ]; then
              return 1  # already-applied
            else
              return 0  # cron up-to-date but queue missing → re-apply queue only
            fi
            ;;
          ts-supabase-edge)
            return 1  # already-applied (no queue-monitor.ts required)
            ;;
        esac
      fi

      if [ "$project" = "$hash_v_19" ]; then
        return 0  # clean-to-apply (canonical match against v1.19.0 baseline)
      fi

      return 2  # DIRTY
    }

    # ─── emit_refuse_artifacts_021 — mirror of 0019's emit_refuse_artifacts_for ─────

    emit_refuse_artifacts_021() {

      # project's cron-monitor.ts and the v1.20.0 template baseline. cf-worker +
      # cf-pages also list would-be queue-monitor.ts.

          # Append the would-be queue-monitor.ts after a separator
          printf '\n--- would-add: queue-monitor.ts ---\n' >> "$dir/.observability-0021.patch"
          cat "$qm_template" >> "$dir/.observability-0021.patch"
          ;;
      esac
    }

    # ─── apply_root_021 — copy updated cron-monitor.ts + new queue-monitor.ts ────────

    apply_root_021() {
      local dir="$1" stack="$2" src="$3"

            info "  (dry-run) would copy: $qm -> $dir/queue-monitor.ts (add)"
          else
            cp "$cm" "$dir/cron-monitor.ts" || return 1
            cp "$qm" "$dir/queue-monitor.ts" || return 1
            info "  migrated: $dir  (stack: $stack) — updated cron-monitor.ts + added queue-monitor.ts (Migration 0021)"

        ts-supabase-edge)

          fi
          ;;
        go-fly-http)
          info "  SKIP: go-fly-http out of scope per Phase 25 D-12"
          ;;
        *)
          info "  SKIP_UNSUPPORTED: $dir  (stack: $stack)"
          ;;
      esac
      return 0
    }

    # ─── version-bump helper (MIRROR of 0019) ────────────────────────────────────────

    bump_version() {
      sed -i.bak -E 's/^(version:[[:space:]]*)1\.19\.0$/\11.20.0/' .claude/skills/agentic-apps-workflow/SKILL.md
      rm -f .claude/skills/agentic-apps-workflow/SKILL.md.bak
    }

    # ─── Main ────────────────────────────────────────────────────────────────────────

    # Discover ROOTS — same pipeline as 0019.
    # ... (mirror discovery code) ...

    if [ ${#ROOTS[@]} -eq 0 ]; then
      info "no materialised observability wrapper found — nothing to migrate."
      bump_version
      exit 0
    fi

    # Phase 1: all-clean-gate.
    DIRTY=()
    CLEAN=()
    ALREADY=()
    for dir in "${ROOTS[@]}"; do
      stack=$(classify_stack "$dir")
      is_clean_to_apply_021 "$dir" "$stack"
      case $? in
        0) CLEAN+=("$dir") ;;
        1) ALREADY+=("$dir") ;;
        2) DIRTY+=("$dir") ;;
      esac
    done

    if [ ${#DIRTY[@]} -gt 0 ]; then
      warn "Migration 0021 REFUSE — ${#DIRTY[@]} wrapper(s) have hand-modified cron-monitor.ts"
      for dir in "${DIRTY[@]}"; do
        stack=$(classify_stack "$dir")
        emit_refuse_artifacts_021 "$dir" "$stack"
        warn "  $dir — see .observability-0021.patch"
      done
      exit 1
    fi

    if [ ${#CLEAN[@]} -eq 0 ] && [ ${#ALREADY[@]} -gt 0 ]; then
      info "all ${#ALREADY[@]} wrapper(s) already at v1.20.0 — SKIP_ALREADY"
      bump_version
      exit 0
    fi

    # Phase 3: apply.
    EXIT_CODE=0
    for dir in "${CLEAN[@]}"; do
      stack=$(classify_stack "$dir")
      src="$(stack_template_dir "$stack")"
      apply_root_021 "$dir" "$stack" "$src" || EXIT_CODE=1
    done

    # ALREADY roots are skipped silently.

    if [ "$EXIT_CODE" -eq 0 ]; then
      bump_version
      info "Migration 0021 complete — version 1.20.0"
    else
      warn "Migration 0021 had errors; version NOT bumped"
    fi
    exit $EXIT_CODE
